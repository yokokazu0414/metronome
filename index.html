修正版の `playTamb` 関数を以下のように変更してください。これにより、Hi-Hat と Tambourine が異なる音色になります。 ```javascript function playTamb(t){ // Tambourine: multiple bright jingle grains (distinct from single-hit Hi-Hat) const grains = 4; for (let k = 0; k < grains; k++) { const delay = (k === 0 ? 0 : (0.010 + Math.random()*0.020) * k); // small stagger const n = ctx.createBufferSource(); const len = (ctx.sampleRate * 0.022) | 0; // very short grain const b = ctx.createBuffer(1, len, ctx.sampleRate); const d = b.getChannelData(0); for (let i = 0; i < d.length; i++) { // tapered noise for a crisp jingle tick d[i] = (Math.random()*2 - 1) * Math.exp(-i / (len * 0.6)); } n.buffer = b; const bp1 = ctx.createBiquadFilter(); bp1.type = 'bandpass'; bp1.frequency.value = 7500 + Math.random()*2000; bp1.Q.value = 8; const bp2 = ctx.createBiquadFilter(); bp2.type = 'bandpass'; bp2.frequency.value = 12000; bp2.Q.value = 3; const g = ctx.createGain(); g.gain.setValueAtTime(0.55, t + delay); g.gain.exponentialRampToValueAtTime(0.001, t + delay + 0.08); n.connect(bp1).connect(bp2).connect(g).connect(master); n.start(t + delay); n.stop(t + delay + 0.09); } } n.buffer=b; const bp=ctx.createBiquadFilter(); bp.type='bandpass'; bp.frequency.value=6000; // Hi-Hat より少し低め bp.Q.value=0.7; const g=ctx.createGain(); g.gain.setValueAtTime(0.8,t); g.gain.exponentialRampToValueAtTime(0.001,t+0.12); n.connect(bp).connect(g).connect(master); n.start(t); n.stop(t+0.13); } ``` これで Tambourine の音は、Hi-Hat（高域カット、短い）とは異なり、中高域でやや長めの残響を持つ音になります。
